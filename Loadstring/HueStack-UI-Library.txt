local DraggableObject = {}
DraggableObject.__index = DraggableObject

function DraggableObject.new(Object)
	local self = {Object = Object, Dragging = false}
	setmetatable(self, DraggableObject)
	return self
end

function DraggableObject:Enable()
	local object = self.Object
	local dragInput, dragStart, startPos, preparingToDrag = nil, nil, nil, false
	local function update(input)
		local delta = input.Position - dragStart
		local newPosition = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
		object.Position = newPosition
		return newPosition
	end
	self.InputBegan = object.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			preparingToDrag = true
			local connection
			connection = input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End and (self.Dragging or preparingToDrag) then
					self.Dragging = false
					connection:Disconnect()
					if self.DragEnded and not preparingToDrag then self.DragEnded() end
					preparingToDrag = false
				end
			end)
		end
	end)
	self.InputChanged = object.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)
	self.InputChanged2 = UserInputService.InputChanged:Connect(function(input)
		if not object.Parent then self:Disable() return end
		if preparingToDrag then
			preparingToDrag = false
			if self.DragStarted then self.DragStarted() end
			self.Dragging = true
			dragStart = input.Position
			startPos = object.Position
		end
		if input == dragInput and self.Dragging then
			local newPosition = update(input)
			if self.Dragged then self.Dragged(newPosition) end
		end
	end)
end

function DraggableObject:Disable()
	self.InputBegan:Disconnect()
	self.InputChanged:Disconnect()
	self.InputChanged2:Disconnect()
	if self.Dragging then
		self.Dragging = false
		if self.DragEnded then self.DragEnded() end
	end
end

local function bindTheme(callback)
	table.insert(themeUpdaters, callback)
	callback()
end

local function applyCorner(instance, radius)
	local corner = instance:FindFirstChild("UICorner") or Instance.new("UICorner")
	corner.CornerRadius = radius or theme.CornerRadius
	corner.Parent = instance
	return corner
end

local function applyStroke(instance, color, transparency)
	local stroke = instance:FindFirstChild("UIStroke") or Instance.new("UIStroke")
	stroke.Color = color or theme.SecondaryColor
	stroke.Thickness = theme.StrokeThickness
	stroke.Transparency = transparency or 0
	stroke.ApplyStrokeMode = Enum.ApplyStrokeMode.Border
	stroke.Parent = instance
	return stroke
end

local function createShadow(parent)
	local shadow = Instance.new("ImageLabel")
	shadow.Name = "Shadow"
	shadow.Image = "rbxassetid://1316045217"
	shadow.ScaleType = Enum.ScaleType.Slice
	shadow.SliceCenter = Rect.new(10, 10, 118, 118)
	shadow.ImageColor3 = theme.ShadowColor
	shadow.ImageTransparency = theme.ShadowTransparency
	shadow.BackgroundTransparency = 1
	shadow.Size = UDim2.new(1, 20, 1, 20)
	shadow.Position = UDim2.new(0, -10, 0, -10)
	shadow.ZIndex = parent.ZIndex - 1
	shadow.Parent = parent
	return shadow
end

local function tween(target, props, duration)
	local t = TweenService:Create(
		target, 
		TweenInfo.new(duration or theme.TransitionSpeed, theme.AnimationEasing, theme.AnimationDirection), 
		props
	)
	t:Play()
	return t
end

function HueStack.SetTheme(newTheme)
	for k, v in pairs(newTheme) do theme[k] = v end
	for _, updater in ipairs(themeUpdaters) do pcall(updater) end
end

function HueStack.GetTheme()
	return table.clone(theme)
end

function HueStack.ResetTheme()
	HueStack.SetTheme(defaultTheme)
end

					trackBase.Size = UDim2.new(1, -16, 0, 6)
					trackBase.Position = UDim2.new(0, 8, 1, -12)
					trackBase.Text = ""
					trackBase.AutoButtonColor = false
					trackBase.Parent = sli
					local fill = Instance.new("Frame")
					fill.Size = UDim2.new(0, 0, 1, 0)
					fill.Parent = trackBase
					local thumb = Instance.new("Frame")
					thumb.Size = UDim2.new(0, 14, 0, 14)
					thumb.Position = UDim2.new(1, -7, 0.5, 0)
					thumb.AnchorPoint = Vector2.new(0.5, 0.5)
					thumb.Parent = fill
					local min, max, current = eSettings.Min or 0, eSettings.Max or 100, eSettings.Default or eSettings.Min or 0
					local dragging = false
					bindTheme(function()
						title.TextColor3 = theme.TextColor valLabel.TextColor3 = theme.MutedTextColor
						trackBase.BackgroundColor3 = theme.BackgroundColor fill.BackgroundColor3 = theme.AccentColor
						thumb.BackgroundColor3 = Color3.new(1,1,1) applyCorner(trackBase, UDim.new(1, 0))
						applyCorner(fill, UDim.new(1, 0)) applyCorner(thumb, UDim.new(1, 0))
					end)
					local function setVal(val)
						current = math.clamp(math.floor(val), min, max)
						valLabel.Text = tostring(current)
						tween(fill, {Size = UDim2.new((current - min) / (max - min), 0, 1, 0)}, 0.1)
						if eSettings.OnChange then eSettings.OnChange(current) end
					end
					setVal(current)
					local function move(input)
						setVal(min + ((max - min) * math.clamp((input.Position.X - trackBase.AbsolutePosition.X) / trackBase.AbsoluteSize.X, 0, 1)))
					end
					trackBase.InputBegan:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = true move(input) end end)
					UserInputService.InputEnded:Connect(function(input) if input.UserInputType == Enum.UserInputType.MouseButton1 then dragging = false end end)
					UserInputService.InputChanged:Connect(function(input) if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then move(input) end end)
					return {SetValue = setVal}

				elseif eType == "Dropdown" then
					local ddContainer = Instance.new("Frame")
					ddContainer.Size = UDim2.new(1, 0, 0, 60)
					ddContainer.BackgroundTransparency = 1
					ddContainer.Parent = secContainer
					local title = Instance.new("TextLabel")
					title.Size = UDim2.new(1, -16, 0, 20)
					title.Position = UDim2.new(0, 8, 0, 0)
					title.BackgroundTransparency = 1
					title.TextXAlignment = Enum.TextXAlignment.Left
					title.Font = theme.FontContent
					title.TextSize = theme.ContentTextSize
					title.Text = eSettings.Text or "Dropdown"
					title.Parent = ddContainer
					local ddBtn = Instance.new("TextButton")
					ddBtn.Size = UDim2.new(1, -16, 0, 36)
					ddBtn.Position = UDim2.new(0, 8, 0, 24)
					ddBtn.Text = ""
					ddBtn.AutoButtonColor = false
					ddBtn.Parent = ddContainer
					local ddText = Instance.new("TextLabel")
					ddText.Size = UDim2.new(1, -40, 1, 0)
					ddText.Position = UDim2.new(0, 10, 0, 0)
					ddText.BackgroundTransparency = 1
					ddText.TextXAlignment = Enum.TextXAlignment.Left
					ddText.Font = theme.FontContent
					ddText.TextSize = theme.ContentTextSize
					ddText.Text = eSettings.Default or "Select..."
					ddText.Parent = ddBtn
					local icon = Instance.new("ImageLabel")
					icon.Size = UDim2.new(0, 16, 0, 16)
					icon.Position = UDim2.new(1, -24, 0.5, 0)
					icon.AnchorPoint = Vector2.new(0, 0.5)
					icon.BackgroundTransparency = 1
					icon.Image = Icons.ChevronDown
					icon.Parent = ddBtn
					
					local dropList = Instance.new("ScrollingFrame")
					dropList.Size = UDim2.new(0, 0, 0, 0)
					dropList.Visible = false
					dropList.ZIndex = 100
					dropList.ScrollBarThickness = 2
					dropList.Parent = gui
					local dropLayout = Instance.new("UIListLayout")
					dropLayout.Padding = UDim.new(0, 2)
					dropLayout.Parent = dropList
					local dropPadding = Instance.new("UIPadding", dropList)
					dropPadding.PaddingTop = UDim.new(0, 4) dropPadding.PaddingBottom = UDim.new(0, 4)
					dropPadding.PaddingLeft = UDim.new(0, 4) dropPadding.PaddingRight = UDim.new(0, 4)
					
					bindTheme(function()
						title.TextColor3 = theme.TextColor ddText.TextColor3 = theme.TextColor
						icon.ImageColor3 = theme.MutedTextColor ddBtn.BackgroundColor3 = theme.BackgroundColor
						dropList.BackgroundColor3 = theme.BackgroundColor dropList.ScrollBarImageColor3 = theme.SecondaryColor
						applyCorner(ddBtn, UDim.new(0, 6)) applyStroke(ddBtn, theme.SecondaryColor)
						applyCorner(dropList, UDim.new(0, 6)) applyStroke(dropList, theme.SecondaryColor)
					end)

					local isOpen = false
					local function toggle()
						isOpen = not isOpen
						tween(icon, {Rotation = isOpen and 180 or 0})
						if isOpen then
							dropList.Position = UDim2.new(0, ddBtn.AbsolutePosition.X, 0, ddBtn.AbsolutePosition.Y + 40)
							dropList.Size = UDim2.new(0, ddBtn.AbsoluteSize.X, 0, 0)
							dropList.Visible = true
							tween(dropList, {Size = UDim2.new(0, ddBtn.AbsoluteSize.X, 0, math.min(#dropList:GetChildren() * 32, 150))})
						else
							tween(dropList, {Size = UDim2.new(0, ddBtn.AbsoluteSize.X, 0, 0)}, 0.15).Completed:Connect(function() dropList.Visible = false end)
						end
					end
					ddBtn.MouseButton1Click:Connect(toggle)
					
					local function pop(opts)
						for _, c in ipairs(dropList:GetChildren()) do if c:IsA("TextButton") then c:Destroy() end end
						for _, opt in ipairs(opts) do
							local oBtn = Instance.new("TextButton")
							oBtn.Size = UDim2.new(1, 0, 0, 30) oBtn.Text = opt
							oBtn.Font = theme.FontContent oBtn.TextSize = theme.ContentTextSize
							oBtn.AutoButtonColor = false oBtn.BackgroundColor3 = theme.BackgroundColor oBtn.TextColor3 = theme.TextColor
							applyCorner(oBtn, UDim.new(0, 4))
							oBtn.Parent = dropList
							oBtn.MouseEnter:Connect(function() tween(oBtn, {BackgroundColor3 = theme.SecondaryColor}) end)
							oBtn.MouseLeave:Connect(function() tween(oBtn, {BackgroundColor3 = theme.BackgroundColor}) end)
							oBtn.MouseButton1Click:Connect(function()
								ddText.Text = opt toggle()
								if eSettings.OnSelect then eSettings.OnSelect(opt) end
							end)
						end
					end
					if eSettings.Options then pop(eSettings.Options) end
					return {
						RefreshOptions = function(_, nOpts, keep)
							pop(nOpts) if not keep then ddText.Text = "Select..." end
						end
					}

				elseif eType == "Input" then
					local inContainer = Instance.new("Frame")
					inContainer.Size = UDim2.new(1, 0, 0, 60)
					inContainer.BackgroundTransparency = 1
					inContainer.Parent = secContainer
					local title = Instance.new("TextLabel")
					title.Size = UDim2.new(1, -16, 0, 20)
					title.Position = UDim2.new(0, 8, 0, 0)
					title.BackgroundTransparency = 1
					title.TextXAlignment = Enum.TextXAlignment.Left
					title.Font = theme.FontContent
					title.TextSize = theme.ContentTextSize
					title.Text = eSettings.Text or ""
					title.Parent = inContainer
					local tb = Instance.new("TextBox")
					tb.Size = UDim2.new(1, -16, 0, 36)
					tb.Position = UDim2.new(0, 8, 0, 24)
					tb.Text = ""
					tb.PlaceholderText = eSettings.Placeholder or "..."
					tb.Font = theme.FontContent
					tb.TextSize = theme.ContentTextSize
					tb.TextXAlignment = Enum.TextXAlignment.Left
					tb.ClearTextOnFocus = false
					tb.Parent = inContainer
					local tbPadding = Instance.new("UIPadding", tb)
					tbPadding.PaddingLeft = UDim.new(0, 10) tbPadding.PaddingRight = UDim.new(0, 10)
					bindTheme(function()
						title.TextColor3 = theme.TextColor tb.TextColor3 = theme.TextColor
						tb.PlaceholderColor3 = theme.MutedTextColor tb.BackgroundColor3 = theme.BackgroundColor
						applyCorner(tb, UDim.new(0, 6)) applyStroke(tb, theme.SecondaryColor)
					end)
					tb.Focused:Connect(function() tween(applyStroke(tb), {Color = theme.AccentColor}) end)
					tb.FocusLost:Connect(function()
						tween(applyStroke(tb), {Color = theme.SecondaryColor})
						if eSettings.OnFocusLost then eSettings.OnFocusLost(tb.Text) end
					end)
					return tb

				elseif eType == "Label" then
					local lbl = Instance.new("TextLabel")
					lbl.Size = UDim2.new(1, -16, 0, 20)
					lbl.BackgroundTransparency = 1
					lbl.TextXAlignment = Enum.TextXAlignment.Left
					lbl.Font = theme.FontContent
					lbl.TextSize = eSettings.TextSize or theme.ContentTextSize
					lbl.Text = eSettings.Text or "Label"
					lbl.Parent = secContainer
					bindTheme(function() lbl.TextColor3 = theme.TextColor end)
					return lbl
				end
			end

			return section
		end
		return tab
	end

	if settings.IncludeThemeEditor then
		local tTab = window:AddTab("Appearance", Icons.Check)
		local tSec = tTab:AddSection({Title = "Theme Colors"})
		local colors = {"AccentColor", "BackgroundColor", "PrimaryColor", "SecondaryColor"}
		for _, c in ipairs(colors) do
			tSec:AddElement("Button", {
				Text = "Randomize " .. c,
				OnClick = function()
					HueStack.SetTheme({[c] = Color3.new(math.random(), math.random(), math.random())})
					HueStack.Notify({Title = "Theme Updated", Content = "Changed " .. c, Type = "Success", Duration = 2})
				end
			})
		end
		tSec:AddElement("Button", {
			Text = "Reset Theme",
			OnClick = function()
				HueStack.ResetTheme()
				HueStack.Notify({Title = "Theme Reset", Content = "Restored to default", Duration = 2})
			end
		})
	end

	return window
end

return HueStack
